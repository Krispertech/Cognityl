import { knowledge, bookingTriggers, greetingTriggers } from './knowledge.js'; /** * Deterministic keyword scoring. No model, no network. * * Latvian diacritics are folded (ā→a, š→s …) so that a caller typing without * diacritics — which is common on phones — still matches. */ const DIACRITICS = { ā: 'a', č: 'c', ē: 'e', ģ: 'g', ī: 'i', ķ: 'k', ļ: 'l', ņ: 'n', š: 's', ū: 'u', ž: 'z', }; export function normalise(text) { if (typeof text !== 'string') return ''; return text .toLowerCase() .replace(/[āčēģīķļņšūž]/g, (c) => DIACRITICS[c] || c) .replace(/[^\p{L}\p{N}\s]/gu, ' ') .replace(/\s+/g, ' ') .trim(); } function matchesAny(haystack, phrases = []) { return phrases.some((p) => haystack.includes(normalise(p))); } /** * Score = number of matching keywords, with longer keywords weighted higher so * that a specific phrase ("kā strādā") beats an incidental short word ("cik"). */ function score(haystack, keywords = []) { let total = 0; for (const kw of keywords) { const needle = normalise(kw); if (needle && haystack.includes(needle)) { total += 1 + needle.length / 20; } } return total; } /** * @returns {{type:'booking'|'greeting'|'answer'|'fallback', id?:string, score:number}} */ export function classify(text, lang = 'lv') { const haystack = normalise(text); if (!haystack) return { type: 'fallback', score: 0 }; // Booking intent wins over informational answers: someone asking to be // contacted should not be handed a pricing paragraph instead. if (matchesAny(haystack, bookingTriggers[lang] || [])) { return { type: 'booking', score: 1 }; } let best = null; for (const entry of knowledge) { const s = score(haystack, entry.keywords[lang] || []); if (s > 0 && (!best || s > best.score)) best = { id: entry.id, score: s }; } if (best) return { type: 'answer', id: best.id, score: best.score }; // Greeting is checked last so "sveiki, cik maksā?" answers the real question. if (matchesAny(haystack, greetingTriggers[lang] || [])) { return { type: 'greeting', score: 1 }; } return { type: 'fallback', score: 0 }; } export function answerFor(id, lang = 'lv') { const entry = knowledge.find((k) => k.id === id); if (!entry) return null; return entry.answer[lang] || entry.answer.lv; }
